# Schema Snapshot — Regeneration Guide

Run this at the start of every sprint, **after** the Part 1 SQL has been
applied in Supabase, to capture the live deployed schema.
Commit `docs/schema_snapshot.sql` before writing any queries.

## Quick run

```bash
cd apps/api
DATABASE_URL="$(grep DATABASE_URL .env | cut -d= -f2-)" \
  python scripts/refresh_schema.py
```

The script writes `../../docs/schema_snapshot.sql` relative to `apps/api/`.

## Script — `apps/api/scripts/refresh_schema.py`

```python
"""Regenerate docs/schema_snapshot.sql from the live deployed database.

Run from apps/api/:
    DATABASE_URL='postgresql://...' python scripts/refresh_schema.py

Writes the snapshot to docs/schema_snapshot.sql at the repo root.
Requires asyncpg (already a project dependency).
PgBouncer-safe: statement_cache_size=0.
"""
import asyncio
import os
import pathlib
import sys

REPO_ROOT = pathlib.Path(__file__).resolve().parents[3]
OUT_FILE = REPO_ROOT / "docs" / "schema_snapshot.sql"


async def main() -> None:
    database_url = os.environ.get("DATABASE_URL")
    if not database_url:
        sys.exit("DATABASE_URL is not set.")

    import asyncpg

    conn = await asyncpg.connect(database_url, statement_cache_size=0, ssl="require")

    # All user tables, ordered by name
    tables = await conn.fetch(
        """
        SELECT table_name
        FROM information_schema.tables
        WHERE table_schema = 'public'
          AND table_type = 'BASE TABLE'
        ORDER BY table_name
        """
    )

    lines: list[str] = [
        "-- Schema snapshot — auto-generated by scripts/refresh_schema.py",
        "-- DO NOT EDIT BY HAND. Regenerate with: cd apps/api && python scripts/refresh_schema.py",
        "",
    ]

    for (table_name,) in tables:
        lines.append(f"-- ===== {table_name} =====")
        cols = await conn.fetch(
            """
            SELECT column_name, data_type, is_nullable, column_default
            FROM information_schema.columns
            WHERE table_name   = $1
              AND table_schema = 'public'
            ORDER BY ordinal_position
            """,
            table_name,
        )
        col_defs = []
        for c in cols:
            nullable = "" if c["is_nullable"] == "YES" else " NOT NULL"
            default = f" DEFAULT {c['column_default']}" if c["column_default"] else ""
            col_defs.append(f"--   {c['column_name']:40s} {c['data_type']}{nullable}{default}")

        # Unique constraints
        uqs = await conn.fetch(
            """
            SELECT kcu.column_name, tc.constraint_name
            FROM information_schema.table_constraints tc
            JOIN information_schema.key_column_usage kcu
              ON tc.constraint_name = kcu.constraint_name
             AND tc.table_schema    = kcu.table_schema
            WHERE tc.table_name   = $1
              AND tc.table_schema = 'public'
              AND tc.constraint_type IN ('UNIQUE', 'PRIMARY KEY')
            ORDER BY tc.constraint_name, kcu.ordinal_position
            """,
            table_name,
        )

        lines.extend(col_defs)
        if uqs:
            by_constraint: dict[str, list[str]] = {}
            for u in uqs:
                by_constraint.setdefault(u["constraint_name"], []).append(u["column_name"])
            for cname, ccols in by_constraint.items():
                lines.append(f"--   CONSTRAINT {cname}: ({', '.join(ccols)})")
        lines.append("")

    await conn.close()

    OUT_FILE.parent.mkdir(parents=True, exist_ok=True)
    OUT_FILE.write_text("\n".join(lines) + "\n")
    print(f"Wrote {OUT_FILE}")


if __name__ == "__main__":
    asyncio.run(main())
```

## After running

```bash
git add docs/schema_snapshot.sql
git commit -m "chore: refresh schema snapshot for sprint N"
git push
```

Then read `docs/schema_snapshot.sql` before writing any SQL in the sprint.
